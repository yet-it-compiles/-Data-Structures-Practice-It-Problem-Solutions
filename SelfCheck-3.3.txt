public class Oops3 {
   public static void main(String[] args) {
        // declaration of variables
        double bubble = 867.5309;
        double x = 10.01;
        // Sending to Method printer
        int z = printer(x, bubble);

        System.out.println("z = " + z);
    }

    public static int printer(double x, double bubble) {
        int z = 5;
        System.out.println("x = " + x + " and y = " + 8.0);
        System.out.println("x = " + x + " and y = " + bubble);
        System.out.println("The value from main is: " + bubble);
        return z;
    }
}
